    Step 1 · Create the database

    In SSMS, create a database called taskdb (or reuse devlaunch), then run this to make the table:

    schema.sql
    CREATE TABLE Tasks (
  Id        INT IDENTITY(1,1) PRIMARY KEY,
  Title     NVARCHAR(200) NOT NULL,
  IsDone    BIT NOT NULL DEFAULT 0,
  CreatedAt DATETIME2 NOT NULL DEFAULT SYSDATETIME()
);

import "dotenv/config";

const config = {
  server: process.env.DB_SERVER,
  database: process.env.DB_NAME,
  user: process.env.DB_USER,
  password: process.env.DB_PASSWORD,
  options: {
    instanceName: process.env.DB_INSTANCE,
    encrypt: true,
    trustServerCertificate: true,   // fine for local dev
  },
};

export const poolPromise = sql.connect(config)
  .then((pool) => { console.log("✅ Connected to MSSQL"); return pool; })
  .catch((err) => { console.error("❌ DB connection failed:", err.message); throw err; });

    Step 3 · The API routes & server

    This file is the heart of the back-end: full CRUD, parameterised queries, proper status codes. Everything from Modules 4 & 5, applied.

    server/routes/tasks.js
    import express from "express";
import { poolPromise, sql } from "../db.js";

const router = express.Router();

// GET /api/tasks  — list all
router.get("/", async (req, res, next) => {
  try {
    const pool = await poolPromise;
    const result = await pool.request()
      .query("SELECT * FROM Tasks ORDER BY CreatedAt DESC");
    res.json(result.recordset);
  } catch (err) { next(err); }
});

// POST /api/tasks  — create
router.post("/", async (req, res, next) => {
  try {
    const { title } = req.body;
    if (!title || !title.trim())
      return res.status(400).json({ error: "title is required" });
    const pool = await poolPromise;
    const result = await pool.request()
      .input("title", sql.NVarChar, title.trim())
      .query("INSERT INTO Tasks (Title) OUTPUT INSERTED.* VALUES (@title)");
    res.status(201).json(result.recordset[0]);
  } catch (err) { next(err); }
});

// PUT /api/tasks/:id  — toggle done / edit
router.put("/:id", async (req, res, next) => {
  try {
    const { isDone, title } = req.body;
    const pool = await poolPromise;
    const result = await pool.request()
      .input("id", sql.Int, Number(req.params.id))
      .input("isDone", sql.Bit, isDone ? 1 : 0)
      .input("title", sql.NVarChar, title ?? null)
      .query(`UPDATE Tasks
              SET IsDone = @isDone,
                  Title = COALESCE(@title, Title)
              OUTPUT INSERTED.*
              WHERE Id = @id`);
    if (!result.recordset.length)
      return res.status(404).json({ error: "Not found" });
    res.json(result.recordset[0]);
  } catch (err) { next(err); }
});

// DELETE /api/tasks/:id
router.delete("/:id", async (req, res, next) => {
  try {
    const pool = await poolPromise;
    await pool.request()
      .input("id", sql.Int, Number(req.params.id))
      .query("DELETE FROM Tasks WHERE Id = @id");
    res.status(204).end();
  } catch (err) { next(err); }
});

export default router;
    server/server.js
    import express from "express";
import cors from "cors";
import "dotenv/config";
import tasksRouter from "./routes/tasks.js";

const app = express();
app.use(cors());
app.use(express.json());

app.use("/api/tasks", tasksRouter);

// central error handler → returns 500 with a message
app.use((err, req, res, next) => {
  console.error(err);
  res.status(500).json({ error: "Something went wrong" });
});

const PORT = process.env.PORT || 4000;
app.listen(PORT, () => console.log(`🚀 API on http://localhost:${PORT}`));

    Run npm run dev in server/. You should see "✅ Connected to MSSQL" and "🚀 API on…". Test it by opening localhost:4000/api/tasks — your starter tasks come back as JSON. The back-end is alive!

    Step 4 · The React front-end

    Open a second terminal in task-manager/ and scaffold the client:

    terminal
    npm create vite@latest client -- --template react
cd client
npm install
npm run dev    // serves on http://localhost:5173

    Replace client/src/App.jsx with this — it ties every React concept together: state, effect, fetch, events, lists, forms.

    client/src/App.jsx
    import { useState, useEffect } from "react";
import "./App.css";

const API = "http://localhost:4000/api/tasks";

export default function App() {
  const [tasks, setTasks] = useState([]);
  const [title, setTitle] = useState("");
  const [loading, setLoading] = useState(true);

  // READ — load tasks once on startup
  useEffect(() => {
    fetch(API)
      .then((res) => res.json())
      .then((data) => { setTasks(data); setLoading(false); })
      .catch(() => setLoading(false));
  }, []);

  // CREATE
  async function addTask(e) {
    e.preventDefault();
    if (!title.trim()) return;
    const res = await fetch(API, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ title }),
    });
    const created = await res.json();
    setTasks([created, ...tasks]);
    setTitle("");
  }

  // UPDATE — toggle done
  async function toggle(task) {
    const res = await fetch(`${API}/${task.Id}`, {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ isDone: !task.IsDone }),
    });
    const updated = await res.json();
    setTasks(tasks.map((t) => (t.Id === updated.Id ? updated : t)));
  }

  // DELETE
  async function remove(id) {
    await fetch(`${API}/${id}`, { method: "DELETE" });
    setTasks(tasks.filter((t) => t.Id !== id));
  }

  if (loading) return <p className="center">Loading…</p>;

  return (
    <div className="app">
      <h1>✅ My Tasks</h1>
      <form onSubmit={addTask}>
        <input
          value={title}
          onChange={(e) => setTitle(e.target.value)}
          placeholder="What needs doing?"
        />
        <button>Add</button>
      </form>

      {tasks.length === 0 && <p className="empty">No tasks yet 🎉</p>}

      <ul>
        {tasks.map((task) => (
          <li key={task.Id} className={task.IsDone ? "done" : ""}>
            <input
              type="checkbox"
              checked={!!task.IsDone}
              onChange={() => toggle(task)}
            />
            <span>{task.Title}</span>
            <button onClick={() => remove(task.Id)}>🗑️</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

    Step 5 · Run the whole stack 🚀

    You need both servers running at once, in two terminals:

    Terminal 1 → server/ → npm run dev → :4000 Terminal 2 → client/ → npm run dev → :5173
    1. Open the app at localhost:5173. Your starter tasks appear — that data came from MSSQL, through Express, into React. 🤯
    2. Add a task. React POSTs to Express → Express INSERTs into MSSQL → returns the new row → React shows it.
    3. Check one off. A PUT updates the IsDone bit in the database.
    4. Delete one, then refresh the page. Notice your changes persist — because they're really in the database now. That's the whole point.
    🔧

    Stuck? The debugging checklist:

    SymptomLikely fix
    "CORS error" in the browser consoleMake sure app.use(cors()) is in server.js and the server is running.
    "Failed to fetch"Is the back-end running on :4000? Is the API URL correct?
    500 errorsCheck the server terminal — the real error is printed there.
    "DB connection failed"Check .env values, that SQL Server is running, and SQL auth is enabled.
    Tasks don't persistYou may be looking at the in-memory version — confirm routes use poolPromise.

    Step 6 · Where to go from here

    You have a real, working full-stack app. Now extend it — this is how you cement skills and build a portfolio. Pick a few:

    🟢 Easier

    • Show a count of remaining tasks
    • Add a "clear completed" button (a DELETE for done tasks)
    • Edit a task's title inline
    • Style it with your own colours

    🟣 Harder

    • Add a Users table & log-in (auth)
    • Filter tabs: All / Active / Done
    • Add due dates & sort by them
    • Deploy it online (Render, Railway, Azure)
    🚀

    Deploying: host the React build as static files, run the Node server on a host like Render or Azure App Service, and use a managed database like Azure SQL. Each is a tutorial of its own — but you now have the foundation to follow them.
